31pub fn download_binaries(build_dir: &Path, info: &ReleaseInfo) -> anyhow::Result<PathBuf> {
32    let architecture = if cfg!(target_arch = "x86_64") {
33        "x64"
34    } else {
35        "arm64"
36    };
37    let to_download = &info.assets.iter().find(|e| {
38        let name = e["name"].as_str().unwrap_or("").to_lowercase();
39
40        // OBS-Studio-30.2.1-Windows.zip
41        name.contains("obs-studio")
42            && (name.contains("windows") || name.contains("full"))
43            && name.contains(".zip")
44            && !name.contains("pdb")
45            && name.contains(architecture)
46    });
47
48    if to_download.is_none() {
49        bail!("No OBS Studio binaries found");
50    }
51
52    let to_download = to_download.unwrap();
53    let url = to_download["browser_download_url"]
54        .as_str()
55        .ok_or(anyhow!("No download url found"))?;
56
57    let download_path = build_dir.join("obs-prebuilt-windows.zip");
58
59    #[cfg(feature = "colored")]
60    println!("Downloading OBS from {}", url.green());
61    let hash = download_file(url, &download_path)?;
62
63    let name = to_download["name"].as_str().unwrap_or("");
64    let checksum = &info.checksums.get(&name.to_lowercase());
65
66    if let Some(checksum) = checksum {
67        if checksum.to_lowercase() != hash.to_lowercase() {
68            bail!("Checksums do not match");
69        } else {
70            #[cfg(feature = "colored")]
71            info!("{}", "Checksums match".on_green());
72        }
73    } else {
74        error!("No checksum found for {}", name);
75    }
76
77    Ok(download_path)
78}

80/// Returns hash
81pub fn download_file(url: &str, path: &Path) -> anyhow::Result<String> {
82    let timeout = Duration::from_secs(60);
83    #[cfg(feature = "colored")]
84    debug!("Downloading OBS binaries from {}", url.green());
85
86    let uri = Uri::try_from(url)?;
87    let mut stream = Stream::connect(&uri, Some(timeout))?;
88
89    stream.set_read_timeout(Some(timeout))?;
90    stream.set_write_timeout(Some(timeout))?;
91
92    stream = Stream::try_to_https(stream, &uri, None)?;
93
94    let res = RequestMessage::new(&uri)
95        .header("Connection", "Close")
96        .header("User-Agent", "cargo-obs-build")
97        .parse();
98    stream.write_all(&res)?;
99
100    // Set up variables
101    let (sender, receiver) = mpsc::channel();
102    let (sender_supp, receiver_supp) = mpsc::channel();
103    let mut raw_response_head: Vec<u8> = Vec::new();
104    let mut buf_reader = BufReader::new(stream);
105
106    // Read from the stream and send over data via `sender`.
107    thread::spawn(move || {
108        buf_reader.send_head(&sender);
109
110        let params = receiver_supp.recv();
111        if params.is_err() {
112            return;
113        }
114
115        let params: Vec<&str> = params.unwrap();
116        //TODO this never exists
117        if params.contains(&"chunked") {
118            let mut buf_reader = ChunkReader::from(buf_reader);
119            buf_reader.send_all(&sender);
120        } else {
121            buf_reader.send_all(&sender);
122        }
123    });
124
125    let deadline = Instant::now() + Duration::from_secs(DEFAULT_REQ_TIMEOUT);
126
127    // Receive and process `head` of the response.
128    raw_response_head.receive(&receiver, deadline)?;
129
130    let response = Response::from_head(&raw_response_head)?;
131    let content_len = response.content_len().unwrap_or(1) as u64;
132    let encoding = response.headers().get("Transfer-Encoding");
133    let mut params = Vec::with_capacity(4);
134
135    if response.status_code().is_redirect() {
136        let location = response.headers().get("Location");
137        if location.is_none() {
138            bail!("No location header found");
139        }
140
141        let location = location.unwrap();
142        return download_file(location, path);
143    }
144
145    if let Some(encode) = encoding {
146        if encode == "chunked" {
147            params.push("chunked");
148        }
149    }
150
151    sender_supp.send(params)?;
152
153    if content_len == 0 {
154        bail!("Content length is 0");
155    }
156
157    #[cfg(feature="cli")]
158    let pb = ProgressBar::new(content_len);
159    #[cfg(feature = "cli")]
160    {
161        let style = ProgressStyle::default_bar()
162            .template("{msg}\n{spinner:.green} [{elapsed_precise}] [{wide_bar:.cyan/blue}] {bytes}/{total_bytes} ({bytes_per_sec}, {eta})")
163            .map_err(|e| anyhow!("Couldn't create style {:#?}", e))?
164            .progress_chars("#>-");
165        pb.set_style(style);
166        pb.set_message("Downloading OBS binaries".to_string());
167    }
168
169    let mut file =
170        File::create(path).or(Err(anyhow!("Failed to create file '{}'", path.display())))?;
171    let mut downloaded: u64 = 0;
172
173    let mut hasher = Sha256::new();
174    loop {
175        let now = Instant::now();
176        let remaining_time = deadline - now;
177
178        let item = receiver.recv_timeout(remaining_time);
179        if let Err(_e) = item {
180            break;
181        }
182
183        let chunk = item?;
184
185        hasher.write_all(&chunk)?;
186        file.write_all(&chunk)
187            .or(Err(anyhow!("Error while writing to file")))?;
188
189        let new = std::cmp::min(downloaded + (chunk.len() as u64), content_len);
190        downloaded = new;
191        #[cfg(feature = "cli")]
192        pb.set_position(new);
193    }
194
195    #[cfg(feature = "cli")]
196    pb.finish_with_message(format!("Downloaded OBS to {}", path.display()));
197    trace!("Hashing...");
198    let _ = stdout().flush();
199    Ok(hex::encode(hasher.finalize()))
200}
