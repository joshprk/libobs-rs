1use super::PropertyCreationInfo;

6#[derive(Debug, Getters, Clone)]

10    description: Option<String>,
11    list_type: ObsComboType,

16#[derive(Debug, Getters, Clone)]
17#[skip_new]
18pub struct ObsListItem {
19    name: String,
20    value: ObsListItemValue,
21    disabled: bool,
22}
23
24#[derive(Debug, Clone)]
25pub enum ObsListItemValue {
26    String(String),
27    Int(i64),
28    Float(f64),
29    Bool(bool),
30    Invalid
31}
32
33impl ObsListItem {
34    fn new(name: String, value: ObsListItemValue, disabled: bool) -> Self {
35        Self {
36            name,
37            value,
38            disabled,
39        }
40    }
41}
42
43impl From<PropertyCreationInfo> for ObsListProperty {
44    fn from(
45        PropertyCreationInfo {
46            name,
47            description,
48            pointer,
49        }: PropertyCreationInfo,
50    ) -> Self {
51        assert_type!(List, pointer);
52
53        let list_type = get_enum!(pointer, list_type, ObsComboType);
54        let format = get_enum!(pointer, list_format, ObsComboFormat);
55
56        let count = unsafe { libobs::obs_property_list_item_count(pointer) };
57        let mut items = Vec::with_capacity(count);
58
59        for i in 0..count {
60            let list_name = unsafe {
61                CStr::from_ptr(libobs::obs_property_list_item_name(pointer, i))
62                    .to_str()
63                    .unwrap_or_default()
64                    .to_string()
65            };
66            let is_disabled = unsafe { libobs::obs_property_list_item_disabled(pointer, i) };
67            let value = match format {
68                ObsComboFormat::Invalid => ObsListItemValue::Invalid,
69                ObsComboFormat::Int => {
70                    let int_val = unsafe { libobs::obs_property_list_item_int(pointer, i) };
71                    ObsListItemValue::Int(int_val)
72                }
73                ObsComboFormat::Float => {
74                    let float_val = unsafe { libobs::obs_property_list_item_float(pointer, i) };
75                    ObsListItemValue::Float(float_val)
76                }
77                ObsComboFormat::String => {
78                    let string_val = unsafe {
79                        CStr::from_ptr(libobs::obs_property_list_item_string(pointer, i))
80                            .to_str()
81                            .unwrap_or_default()
82                            .to_string()
83                    };
84                    ObsListItemValue::String(string_val)
85                }
86                ObsComboFormat::Bool => {
87                    let bool_val = unsafe { libobs::obs_property_list_item_bool(pointer, i) };
88                    ObsListItemValue::Bool(bool_val)
89                }
90            };
91            items.push(ObsListItem::new(list_name, value, is_disabled));
92        }
93
94        Self {
95            name,
96            description,
97            list_type,
98            format,
99            items,
100        }
101    }
102}
